Url parser
----------

  >>> from iw.thumbs.url import default_parser
  >>> parser, r = default_parser()
  >>> print parser('/thumbs/10x10/my/album/pic1.jpg')
  ('/my/album/pic1.jpg', (10, 10))

  >>> print parser('/thumbs/10x10//my/album/pic1.jpg')
  ('/my/album/pic1.jpg', (10, 10))

  >>> print parser('//thumbs//10x10//my/album/pic1.jpg')
  ('/my/album/pic1.jpg', (10, 10))
