>>> from indra.base import llsd
>>> llsd.format_xml({'a':3})
'<?xml version="1.0" ?><llsd><map><key>a</key><integer>3</integer></map></llsd>'

>>> llsd.format_xml({'a':[1,2,3,4]})
'<?xml version="1.0" ?><llsd><map><key>a</key><array><integer>1</integer><integer>2</integer><integer>3</integer><integer>4</integer></array></map></llsd>'

>>> a=llsd.format_xml({'a':[1,2,3,4]})
>>> llsd.parse(a)
{'a': [1, 2, 3, 4]}
