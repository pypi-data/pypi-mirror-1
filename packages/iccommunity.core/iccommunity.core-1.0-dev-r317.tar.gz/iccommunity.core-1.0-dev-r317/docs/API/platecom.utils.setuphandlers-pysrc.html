 44

 46        False

 48        >>> from iccommunity.core.setuphandlers import ensure_site 
 49        >>> ensure_site(om) 
 50        >>> ISite.providedBy(om) 
 51        True 
 52

 54    if not IPossibleSite.providedBy(context): 
 55        if hasattr(context, 'getPhysicalPath'):

 61        else:
